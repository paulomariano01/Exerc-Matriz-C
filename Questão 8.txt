#include <stdio.h>
#include <stdlib.h>
#include <time.h>

#define TAMANHO 5


/*
 * Fa√ßa um algoritmo em C que preencha dois vetores A e B (TAMANHO: 5) com valores inteiros.
 * Depois de preencher os vetores A e B, exiba um vetor C, sendo que C = A + B. Usar o RAND.
 */

int main(void) {

    srand(time(NULL));

    int vetorA[TAMANHO], vetorB[TAMANHO], vetorC[TAMANHO];

    for (int i = 0; i < 5; i++) {
        vetorA[i] = rand() % 50;
        vetorB[i] = rand() % 50;
        vetorC[i] = vetorA[i] + vetorB[i];


        printf("Vetor C : %i \n", vetorC[i]);

    }


}
